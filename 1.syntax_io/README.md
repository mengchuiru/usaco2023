# 例题

#### hello world
``` c++
#include <iostream>

using namespace std;

int main()
{
    cout << "hello world!" <<endl;
    return 0;
}
```

#### 输出第二个整数
``` c++
#include <iostream>

using namespace std;

int main()
{
    int a1, a2, a3;
    cin >> a1 >> a2 >> a3;

    cout << a2 <<endl;
    return 0;
}
```

#### 对齐输出
 ``` c++
 #include <cstdio>

int main()
{
    int a1, a2, a3;
    scanf("%d %d %d", &a1, &a2, &a3);

    printf("%8d\n%8d\n%8d", a1, a2, a3);

    return 0;
}
 ```
 #### 保留3位小数
``` c++
#include <cstdio>
int main()
{
    float a;
    scanf("%f", &a);
    printf("%.3f", a);

    return 0;
}
```
#### 字符三角形

``` c++
#include <iostream>
using namespace std;

int main()
{
    char c;
    cin >> c;
    cout << ' ' << ' ' << c << endl;
    cout << ' ' << c << c << c << endl;
    cout << c << c << c << c << c << endl;

    return 0;
}
```
``` c++
#include <cstdio>

int main()
{
    char c;
    scanf("%c", &c);
    printf("  %c\n %c%c%c\n%c%c%c%c%c\n",c,c,c,c,c,c,c,c,c);

    return 0;
}

```
